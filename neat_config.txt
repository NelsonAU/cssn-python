# NEAT configuration for Compositional Sound Synthesis Networks (CSSNs)
#
# Reference: Jónsson, Hoover, Risi — "Interactively Evolving Compositional
#            Sound Synthesis Networks", GECCO 2015.
#
# Parameters taken directly from Section 3.1 or 4.1 are labelled [PAPER].
# Anything we had to choose (even if sometimes the choice is obvious) is
# labelled [NOT IN PAPER].

[NEAT]
# [PAPER] "Population size is 10 per generation"
pop_size               = 10
# [NOT IN PAPER]
fitness_criterion      = max
reset_on_extinction    = False
# [NOT IN PAPER] IEC is user-terminated (fixed generation count via pop.run),
# so fitness-threshold termination is disabled entirely.
no_fitness_termination = True
fitness_threshold      = 999.0

[DefaultGenome]
# Activation functions
# [PAPER] "Gaussian, Bipolar sigmoid, sine and linear, each with a 0.25
#          probability of being added"
activation_options     = gaussian bipolar_sigmoid sin linear
# [NOT IN PAPER] Default for new nodes; paper implies equal probability.
activation_default     = sin
# [NOT IN PAPER] Small rate allows existing nodes to drift to a new function.
activation_mutate_rate = 0.05

# Aggregation
# [NOT IN PAPER] Weighted sum is standard for CPPNs.
aggregation_default     = sum
aggregation_options     = sum
aggregation_mutate_rate = 0.0

# Bias
# [NOT IN PAPER]
bias_init_mean    = 0.0
bias_init_stdev   = 1.0
bias_max_value    = 30.0
bias_min_value    = -30.0
bias_mutate_power = 0.5
# [PAPER] "probability of changing a weight is 0.7" — applied to bias too.
bias_mutate_rate  = 0.7
# [NOT IN PAPER]
bias_replace_rate = 0.1

# Speciation compatibility
# [NOT IN PAPER] Standard neat-python defaults.
compatibility_disjoint_coefficient = 1.0
compatibility_weight_coefficient   = 0.5

# Connection add / delete
# [PAPER] "probability of adding a new connection … 0.13"
conn_add_prob    = 0.13
# [NOT IN PAPER]
conn_delete_prob = 0.05

# Connection enable / disable
# [NOT IN PAPER]
enabled_default     = True
enabled_mutate_rate = 0.01

# Network topology
# [NOT IN PAPER] Feed-forward CPPNs are used in related papers.
feed_forward = True
# [NOT IN PAPER] NEAT starts with a minimal fully-connected topology
#                (inputs wired directly to outputs, no hidden nodes).
initial_connection = full_direct

# Node add / delete
# [PAPER] "probability of adding … a new node … 0.13"
node_add_prob    = 0.13
# [NOT IN PAPER]
node_delete_prob = 0.05

# Network I/O
# [PAPER] "The CSSN has three inputs and two outputs."
#   Input 0 : abs(y)          — waveform coordinate, absolute value
#   Input 1 : sin(n · |y|)   — periodic repetition input
# Note: The paper's third input (bias 1.0) is dropped: neat-python already adds
# a trainable per-node bias term, so including it as an explicit input
# over-parameterises the network. [NOT IN PAPER]
num_inputs  = 2
# [PAPER] "two outputs … carrier oscillator … frequency modulation"
num_outputs = 2
num_hidden  = 0

# Response (output gain per node)
# [NOT IN PAPER]
response_init_mean    = 1.0
response_init_stdev   = 0.0
response_max_value    = 30.0
response_min_value    = -30.0
response_mutate_power = 0.0
response_mutate_rate  = 0.0
response_replace_rate = 0.0

# Weights
# [NOT IN PAPER]
weight_init_mean    = 0.0
weight_init_stdev   = 1.0
weight_max_value    = 30.0
weight_min_value    = -30.0
weight_mutate_power = 0.5
# [PAPER] "probability of changing a weight is 0.7"
weight_mutate_rate  = 0.7
# [NOT IN PAPER]
weight_replace_rate = 0.1

[DefaultSpeciesSet]
# [NOT IN PAPER] Standard neat-python default.
compatibility_threshold = 3.0

[DefaultStagnation]
# [NOT IN PAPER]
species_fitness_func = max
max_stagnation       = 15
species_elitism      = 2

[DefaultReproduction]
# [NOT IN PAPER]
elitism            = 2
survival_threshold = 0.2
